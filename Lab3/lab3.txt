ORG 0000H
LJMP MAIN
ORG 0013H
LJMP INT

ORG 0150H
DIV_LOOP:
	JNZ NO_ZERO
	MOV A,R3
	JB P3.0, NO_ZERO
	DEC A
	SETB P3.0
	SJMP END_DIV
NO_ZERO:CLR P3.0
	DEC A
END_DIV:RET

ORG 0300H
MAIN:	LCALL CONFIGURE_LCD
	LCALL KEYBOARD
	LCALL SEND_DATA
	SUBB A, #30H
	MOV R3,A
	CLR A
	MOV TCON, #85H
	SETB EA
	SETB EX1
	CLR P3.0
	SJMP $

INT:	LCALL DIV_LOOP
	RETI
